                        &\frac{|(-1)^{n+1} * \frac{(2(n+1))!}{(1-2(n+1))4^{n+1}((n+1)!)^{2}}* x^{n+1}|}{|(-1)^{n} * \frac{(2n)!}{(1-2n)4^{n}(n!)^{2}}* x^{n}|}

                                                &= (2n+2)(2n+1)

                                              |x|
